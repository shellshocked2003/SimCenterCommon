149 InputWidgetOpenSeesAnalysis::outputToJSON(QJsonObject &jsonObject)
150 {
151  bool result = true;
152  jsonObject["Application"] = "OpenSees-Simulation";
153  jsonObject["integration"]=theIntegration->text();
154  jsonObject["algorithm"]=theAlgorithm->text();
155  jsonObject["convergenceTest"]=theConvergenceTest->text();
156 
157  QString tolText = theTolerance->text();
158  bool ok;
159  double tolDouble = tolText.QString::toDouble(&ok);
160  if (ok == true)
161  jsonObject["tolerance"]=tolDouble;
162  else
163  jsonObject["tolerance"]= QString("RV.") + tolText;
164 
165  QString dampText = dampingRatio->text();
166  double dampDouble = dampText.QString::toDouble(&ok);
167  if (ok == true)
168  jsonObject["dampingRatio"]=dampDouble;
169  else
170  jsonObject["dampingRatio"]= QString("RV.") + dampText;
171 
172  if (!file->text().isEmpty() && !file->text().isNull()) {
173  QFileInfo fileInfo(file->text());
174  jsonObject["fileName"]= fileInfo.fileName();
175  jsonObject["filePath"]=fileInfo.path();
176  }
177 
178  return result;
179 }

183 InputWidgetOpenSeesAnalysis::inputFromJSON(QJsonObject &jsonObject)
184 {
185  bool result = true;
186  this->clear();
187 
188  if (jsonObject.contains("integration") && jsonObject.contains("algorithm")
189  && jsonObject.contains("convergenceTest")) {
190 
191  theAlgorithm->setText(jsonObject["algorithm"].toString());
192  theConvergenceTest->setText(jsonObject["convergenceTest"].toString());
193  theIntegration->setText(jsonObject["integration"].toString());
194 
195  } else {
196  emit sendErrorMessage("ERROR: InputWidgetOpenSeesAnalysis - no \"integration\" ,\"convergenceTest\" or \"algorithm\" data");
197  return false;
198  }
199 
200  if (jsonObject.contains("tolerance")) {
201  QJsonValue theValue = jsonObject["tolerance"];
202  if (theValue.isString()) {
203  QString text = theValue.toString();
204  text.remove(0,3); // remove RV.
205  theTolerance->setText(text);
206  } else if (theValue.isDouble())
207  theTolerance->setText(QString::number(theValue.toDouble()));
208  } else {
209  emit sendErrorMessage("ERROR: InputWidgetOpenSeesAnalysis - no \"tolerance\" data");
210  return false;
211  }
212 
213  if (jsonObject.contains("dampingRatio")) {
214  QJsonValue theValue = jsonObject["dampingRatio"];
215  if (theValue.isString()) {
216  QString text = theValue.toString();
217  text.remove(0,3); // remove RV.
218  dampingRatio->setText(text);
219  } else if (theValue.isDouble())
220  dampingRatio->setText(QString::number(theValue.toDouble()));
221  } else {
222  emit sendErrorMessage("ERROR: InputWidgetOpenSeesAnalysis - no \"dampingRatio\" data");
223  return false;
224  }
225 
226  if (jsonObject.contains("fileName")) {
227  QJsonValue theName = jsonObject["fileName"];
228  QString fileName = theName.toString();
229  if (jsonObject.contains("filePath")) {
230  QJsonValue theName = jsonObject["filePath"];
231  QString filePath = theName.toString();
232  file->setText(QDir(filePath).filePath(fileName));
233  } else
234  return false;
235  }
236 
237 
238  return result;
239 }
